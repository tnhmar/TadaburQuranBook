%% ── نصف الحزب divider ────────────────────────────────────────
%% Usage: \nisfhizb{1}{1}{سورة الفاتحة ١–٧ + البقرة ١–٥}
%%   #1 = hizb number (1..60)
%%   #2 = nisf (1=first half, 2=second half)
%%   #3 = surah:ayah range description
\newcommand{\nisfhizb}[3]{%
  \begin{hizbbox}{نصف الحزب~#2 — الحزب~#1\quad|\quad\small #3}%
    \centering
    {\quranfont\large
      \ifnum#2=1 ۞\else ۝\fi  %% ۞ = hizb mark, ۝ = quarter
    }\par
