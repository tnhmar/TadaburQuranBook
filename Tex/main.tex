\newcommand{\makehalftitle}{%

\newcommand{\makecover}{%
  \begin{titlepage}\thispagestyle{empty}\vspace*{2cm}
    \begin{center}
      {\large\scshape\theBookVolume\par}\vspace{0.3cm}
      {\small\itshape\theBookVolumeScope\par}\vspace{1.2cm}
      \hrule height 0.5pt\par\vspace{1.2cm}
      {\fontsize{38}{46}\bfseries\selectfont\theBookTitle\par}
      \vspace{1.2cm}\hrule height 0.5pt\par\vspace{0.8cm}
      {\fontsize{11}{14}\selectfont\theBookSubtitle\par}
      \vspace{1.5cm}{\Large\theBookAuthor\par}\vfill
      {\normalsize\theBookDate\par}\vspace{0.5cm}
    \end{center}
  \end{titlepage}\nopagecolor}
